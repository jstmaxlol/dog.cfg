29 inline Config Load(const std::string& filename) {
30 std::ifstream file(filename);
31 Config config;
32 std::string line, current_section;
33
34 while(std::getline(file, line)) {
35 line.erase(0, line.find_first_not_of(" \t\r\n"));
36 line.erase(line.find_last_not_of(" \t\r\n") + 1);
37 if (line.empty() || line[0] == '#' || line[0] == ';') continue;
38
39 if (line.front() == '[' && line.back() == ']') {
40 current_section = line.substr(1, line.size() - 2);
41 } else {
42 auto sep = line.find('=');
43 if (sep != std::string::npos) {
44 std::string key = line.substr(0, sep);
45 std::string val = line.substr(sep + 1);
46 auto trim = [](std::string& s) {
47 s.erase(0, s.find_first_not_of(" \t\r\n"));
48 s.erase(s.find_last_not_of(" \t\r\n") + 1);
49 };
50 trim(key);
51 trim(val);
52 config[current_section][key] = val;
53 }
54 }
55 }
56 return config;
57 }

117 inline T Get(const Config& config, const std::string& section, const std::string& key, T fallback = T()) {
118 auto sec_it = config.find(section);
119 if (sec_it == config.end()) return fallback;
120 auto key_it = sec_it->second.find(key);
121 if (key_it == sec_it->second.end()) return fallback;
122
123 std::istringstream iss(key_it->second);
124 T value;
